Basic SQL Commands

SELECT Queries

SELECT column1, column2 FROM table_name WHERE condition;

INSERT Statement

INSERT INTO table_name (column1, column2) VALUES (value1, value2);

UPDATE Statement

UPDATE table_name SET column1 = value1, column2 = value2 WHERE condition;

DELETE Statement

DELETE FROM table_name WHERE condition;

Pro Tip

Always use WHERE clause when updating or deleting records to avoid accidental modifications.

Advanced SQL Queries

Subqueries

SELECT name FROM employees WHERE department_id IN ( SELECT id FROM departments WHERE location = 'NY' );

DROP COLUMN

ALTER TABLE table_name DROP COLUMN column_name;

DELETE Rows with NULL Values

DELETE FROM students WHERE username IS NULL;

UPDATE Rows with NULL Values

UPDATE students SET username = 'DefaultUser' WHERE username IS NULL;

SELECT All Rows

SELECT * FROM students;

Database Administration

User Management

CREATE USER 'username'@'localhost' IDENTIFIED BY 'password'; GRANT SELECT, INSERT ON database.* TO 'username'@'localhost';

ALTER TABLE - Add UNIQUE Constraint

ALTER TABLE students ADD UNIQUE(username);

ALTER TABLE - Drop Index

ALTER TABLE students DROP INDEX username;
